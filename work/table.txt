<?php
error_reporting(-1);
ini_set('display_errors','on');
header('Content-Type: text/html; charset=utf-8');

echo "<link rel='stylesheet' href='table-style.css'>";

echo '<table class="my-table"><tr>';
for ($x = 1; $x <= 3; ++$x)
{
	echo '<tr>';
	for ($y = 1; $y <= 5; ++$y)
	{
		if ($x == 1 && $y == 2) {
			echo '<td class="bg-yellow">'.$x.':'.$y.'</td>';
		}

		elseif ($x == 2 && ($y == 3 || $y == 4)) {
			echo '<td class="bg-green">'.$x.':'.$y.'</td>';
		}

		elseif ($x == 3) {
			echo '<td class="bg-red">'.$x.':'.$y.'</td>';
		}

		else {
			echo '<td class="my-td">'.$x.':'.$y.'</td>';
		}
	}
}
echo '</tr></table>';
